<script>
    document.addEventListener("DOMContentLoaded", function(event) {
        function addDraggable(width, height, content) {
            var $draggable = $(
                '<div class="wall__draggable" style="width: ' + width + 'px; height: ' + height + 'px;">' +
                '<div class="wall__draggable-inner">' +
                '<p class="wall__draggable-content">' + content + '</p>' +
                '</div>' +
                '</div>');
            // draggable
            $(".wall__container").append($draggable);
            //make it "draggable" and "resizable"
            $draggable.draggable({
                grid: [5, 5],
                scroll: true,
                containment: "parent"
            });
        }
        $('.button').click(function() {
            addDraggable($(this).attr('data-draggable-width'), $(this).attr('data-draggable-height'), $(this).attr('id'));
        });
    });
</script>
